3   /**
4    * Luokka tarjoaa metodeita pelaajan pisteiden laskemiseen ja pelin päätyttyä
5    * ilmestyvään viestiin.
6    *
7    */
8   public class Pelaaja {
9   
10      private String nimi;
11      private int pisteet;
12      private int edellisetPisteet;
13      private int bonuspisteet;
14  
15      public Pelaaja(String nimi) {
16          this.nimi = nimi;
17          this.pisteet = 0;
18          this.bonuspisteet = 0;
19      }
20  
21      public String getNimi() {
22          return nimi;
23      }
24  
25      public int getPisteet() {
26          return pisteet;
27      }
28  
29      /**
30       * Pistelaskumetodi huolehtii pelaajan pisteiden kirjaamisesta. Pistelasku
31       * menee niin, että löydetystä parista saa kaksi pistettä. Jos paria ei
32       * löytynyt, pelaaja saa miinuspisteen. Kahden parin löytämisestä peräkkäin
33       * saa kaksi bonuspistettä.
34       *
35       * @param pisteet muistipelin välittämä pistemäärä, joka on joko 2 parin
36       * löytämisestä tai -1 hudista.
37       *
38       */
39      public void kirjaaPisteet(int pisteet) {
40  
41          if (edellisetPisteet == 2 && pisteet == 2) {
42              this.pisteet = this.pisteet + pisteet + 2;
43              this.bonuspisteet = this.bonuspisteet + 2;
44  
45          } else {
46  
47              this.pisteet = this.pisteet + pisteet;
48          }
49          edellisetPisteet = pisteet;
50      }
51  
52      /**
53       * Metodi palauttaa pelaajan pelin aikana keräämien bonuspisteiden määrän.
54       *
55       */
56      public int bonuspisteet() {
57  
58          return bonuspisteet;
59      }
60  
61      /**
62       * Metodissa määritellään viesti, jonka pelaaja saa pelin lopussa. Viesti
63       * vaihtelee sen mukaan, kuinka hyvin pelaaja pärjäsi pelissä eli kerättyjen
64       * pisteiden ja bonuspisteiden mukaan.
65       *
66       * @return
67       */
68      public String loppuViesti() {
69          String viesti = "";
70  
71          if (pisteet > 0 && bonuspisteet > 0) {
72              viesti = "Huikea suoritus " + nimi + ", olet varsinainen muistiguru!"
73                      + "\nKeräsit bonuspisteitä " + bonuspisteet
74                      + " ja lopullinen pistesaldosi on " + pisteet
75                      + " pistettä!"
76                      + "\nLoistavaa!";
77          }
78          if (pisteet > 0 && bonuspisteet == 0) {
79              viesti = "Hieno suoritus " + nimi + ", olet varsin etevä muistaja!"
80                      + "\nBonuspisteet jäivät tällä kertaa saamatta."
81                      + "\nKeräsit siitä huolimatta kuitenkin " + pisteet
82                      + " pistettä! Loistavaa!";
83          }
84          if (pisteet <= 0 && bonuspisteet > 0) {
85              viesti = "Hyvä " + nimi + ", löysit kaikki parit!"
86                      + "\nPistesaldosi ei tällä kertaa ihan yltänyt plussan puolelle, sait yhteensä "
87                      + pisteet + " pistettä."
88                      + "\nKeräsit kuitenkin hienot " + bonuspisteet
89                      + " bonuspistettä!";
90          }
91          if (pisteet <= 0 && bonuspisteet == 0) {
92              viesti = "Hyvä " + nimi + ", löysit kaikki parit!"
93                      + "\nPistesaldosi ei tällä kertaa ihan yltänyt plussan puolelle, sait yhteensä "
94                      + pisteet + " pistettä."
95                      + "\nParempi onni ensi kerralla, eikun treenailemaan lisää!";
96          }
97  
98          return viesti;
99  
100     }
