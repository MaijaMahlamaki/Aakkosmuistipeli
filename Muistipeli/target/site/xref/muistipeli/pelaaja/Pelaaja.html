1   package muistipeli.pelaaja;
2   
3   import java.text.DecimalFormat;
4   
5   /**
6    * Luokka tarjoaa metodeita pelaajan pisteiden laskemiseen ja pelin päätyttyä
7    * ilmestyvään viestiin.
8    *
9    */
10  public class Pelaaja {
11  
12      private String nimi;
13      private int pisteet;
14      private int edellisetPisteet;
15      private int bonuspisteet;
16      private int kierrokset;
17      private int yhteispisteet;
18      private double keskiarvo;
19  
20      /**
21       * Pelaaja saa konstruktorin parametrina pelaajan nimen. Konstruktorissa
22       * asetetaan pisteet, bonuspisteet, kierrokset, yhteispisteet ja kierrosten
23       * pisteiden keskiarvo alkamaan nollasta.
24       *
25       * @param nimi pelaajan nimi
26       */
27      public Pelaaja(String nimi) {
28          this.nimi = nimi;
29          this.pisteet = 0;
30          this.bonuspisteet = 0;
31          this.kierrokset = 0;
32          this.yhteispisteet = 0;
33          this.keskiarvo = 0.0;
34      }
35  
36      public String getNimi() {
37          return nimi;
38      }
39  
40      public int getPisteet() {
41          return pisteet;
42      }
43  
44      /**
45       * Pistelaskumetodi huolehtii pelaajan pisteiden kirjaamisesta. Pistelasku
46       * menee niin, että löydetystä parista saa kaksi pistettä. Jos paria ei
47       * löytynyt, pelaaja saa miinuspisteen. Kahden parin löytämisestä peräkkäin
48       * saa kaksi bonuspistettä. Metodi päivittää aina samalla myös pelien
49       * yhteispisteet, jotta pelaajalle voidaan näyttää pelin lopuksi tämän ja
50       * mahdollisten aiempien pelien yhteispisteet ja pistekeskiarvo.
51       *
52       * @param pisteet muistipelin välittämä pistemäärä, joka on joko 2 parin
53       * löytämisestä tai -1 hudista.
54       *
55       */
56      public void kirjaaPisteet(int pisteet) {
57  
58          if (edellisetPisteet == 2 && pisteet == 2) {
59              this.pisteet = this.pisteet + 2;
60              this.bonuspisteet = this.bonuspisteet + 2;
61              this.yhteispisteet = this.yhteispisteet + 2 + 2;
62  
63          } else {
64  
65              this.pisteet = this.pisteet + pisteet;
66              this.yhteispisteet = this.yhteispisteet + pisteet;
67          }
68          edellisetPisteet = pisteet;
69      }
70  
71      /**
72       * Metodi palauttaa pelaajan pelin aikana keräämien bonuspisteiden määrän.
73       *
74       * @return bonuspisteet
75       */
76      public int bonuspisteet() {
77  
78          return bonuspisteet;
79      }
80  
81      /**
82       * Metodissa määritellään viesti, jonka pelaaja saa pelin lopussa. Viesti
83       * vaihtelee sen mukaan, kuinka hyvin pelaaja pärjäsi pelissä eli kerättyjen
84       * pisteiden ja bonuspisteiden mukaan.
85       *
86       * @return kierroksen pisteiden mukaan räätälöity viesti
87       */
88      public String loppuViesti() {
89          String viesti = "";
90  
91          if (pisteet > 4 && bonuspisteet > 4) {
92              viesti = "Huikea suoritus " + nimi + ", olet varsinainen muistiguru!"
93                      + "\nKeräsit " + pisteet
94                      + " pistettä ja " + bonuspisteet
95                      + " bonuspistettä!"
96                      + "\nLoistavaa!"
97                      + "\nPelatut kierrokset: " + kierrokset
98                      + "\nYhteispisteet: " + yhteispisteet
99                      + "\nPisteiden keskiarvo: " + keskiArvo();
100         } else if (pisteet > 2 && bonuspisteet > 2) {
101             viesti = "Hieno suoritus " + nimi + ", olet varsin etevä muistaja!"
102                     + "\nKeräsit " + pisteet
103                     + " pistettä ja " + bonuspisteet
104                     + " bonuspistettä! Loistavaa!"
105                     + "\nPelatut kierrokset: " + kierrokset
106                     + "\nYhteispisteet: " + yhteispisteet
107                     + "\nPisteiden keskiarvo: " + keskiArvo();
108         } else if (pisteet < 0 && bonuspisteet > 2) {
109             viesti = "Hyvä " + nimi + ", löysit kaikki parit!"
110                     + "\nPistesaldosi ei tällä kertaa ihan yltänyt plussan puolelle, sait yhteensä "
111                     + pisteet + " pistettä, mutta keräsit kuitenkin " + bonuspisteet + " bonuspistettä."
112                     + "\nPelatut kierrokset: " + kierrokset
113                     + "\nYhteispisteet: " + yhteispisteet
114                     + "\nPisteiden keskiarvo: " + keskiArvo();
115         } else if (pisteet < 0 && bonuspisteet == 2) {
116             viesti = "Hyvä " + nimi + ", löysit kaikki parit!"
117                     + "\nPistesaldosi ei tällä kertaa ihan yltänyt plussan puolelle, sait yhteensä "
118                     + pisteet + " pistettä ja "
119                     + bonuspisteet + " bonuspistettä."
120                     + "\nParempi onni ensi kerralla, eikun treenailemaan lisää!"
121                     + "\nPelatut kierrokset: " + kierrokset
122                     + "\nYhteispisteet: " + yhteispisteet
123                     + "\nPisteiden keskiarvo: " + keskiArvo();
124         }
125 
126         return viesti;
127 
128     }
129 
130     /**
131      * Metodi lisaa pelaajalle yhden pelikierroksen, jotta pelaajalle voidaan
132      * ilmoittaa pelin lopuksi, montako kierrosta hän on pelannut ja laskea
133      * kaikkien pelien pistekeskiarvo.
134      */
135     public void lisaaKierros() {
136         this.kierrokset = this.kierrokset + 1;
137     }
138 
139     /**
140      * Metodi palauttaa pelaajan pelaamien kierrosten lukumäärän.
141      *
142      * @return kierrosten lukumäärä
143      */
144     public int kierrokset() {
145         return this.kierrokset;
146     }
147 
148     /**
149      * Metodi palauttaa kaikkien pelaajan pelaamien kierrosten pisteiden summan.
150      *
151      * @return yhteispisteet
152      */
153     public int yhteisPisteet() {
154         return yhteispisteet;
155     }
156 
157     /**
158      * Metodi laskee yhteispisteiden ja kierrosten lukumäärän perusteella
159      * pelattujen kierrosten pistekeskiarvon.
160      *
161      * @return keskiarvo
162      */
163     public double keskiArvo() {
164         this.keskiarvo = (double) this.yhteispisteet / this.kierrokset;
165         return keskiarvo;
166     }
167 
168     /**
169      * Metodilla nollataan pisteet kierrosten välillä, jotta seuraavan
170      * kierroksen pistelasku aloitetaan aina nollasta sekä normaaleilla että
171      * bonuspisteillä.
172      */
173     public void nollaaPisteet() {
174         this.pisteet = 0;
175         this.bonuspisteet = 0;
176     }
177 }
